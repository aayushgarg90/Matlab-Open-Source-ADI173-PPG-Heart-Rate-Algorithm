1function [PPG_struct] = PPG_algorithm(PPG_struct)
  2%#codegen
  3
  4%% This function process Raw PPG sensor Data to determin a PulseRate  
  5%
  6% This algorithm is compatible with 
  7%             Version **  
  8%      of the PPG embedded code
  9
 10% Version 12 removed interpolation from peak detector quadrants 1 and 3.
 11%           it was incorrect to use linear interp for peak detection.
 12% 4/14/2015 No longer using Version numbers in the function name(filename)
 13%           
 14%
 15% Structure used - For Reference -
 16% PPG_struct.PulseRate               = uint8(0);
 17% PPG_struct.FilteredDataInput       = single(0);
 18% PPG_struct.PreviousInputVal        = single(0);
 19% PPG_struct.Quadrant                = uint8(0);
 20% PPG_struct.PreviousQTime           = zeros([4,1],'single');
 21% PPG_struct.LocalHigh               = single(0);
 22% PPG_struct.LocalLow                = single(0);
 23% PPG_struct.Threshold               = single(0);
 24% PPG_struct.Timestamp               = uint16(0);
 25% PPG_struct.PreviousTimestamp       = uint16(0);
 26
 27%Knobs
 28%global Knob_Threshold_Goal Knob_ThresholdTC 
 29
 30%Constants
 31k_TimeStampsPerMin      = single(32000 * 60);   % Timestamp resolution is 31.25us (32KHz)
 32k_Max_Timestamp         = uint16(65535);        % for unwrapping overflowed Timestamps
 33
 34%Local variables
 35Interpolated_zerotime       = single(0);
 36PPG_struct.PulseRate        = uint8(0);
 37
 38
 39
 40
 41%%   P U L S E   D E T E C T O R   S T A T E   M A N A G E M E N T %
 42if (PPG_struct.Quadrant == 1) 
 43    % This Quadrant began at the rising zero crossing of systolic pressure
 44    % This PPG_struct.Quadrant ends at the peak systolic pressure
 45    % when peak is detected quadrant changes to, and HPR reported as Q2
 46    if (PPG_struct.FilteredDataInput > PPG_struct.Threshold) 
 47        if (PPG_struct.FilteredDataInput > PPG_struct.LocalHigh) 
 48            PPG_struct.LocalHigh = PPG_struct.FilteredDataInput;
 49        else 
 50            % % Peak Detected
 51            
 52            if (PPG_struct.PreviousTimestamp > PPG_struct.Timestamp)  % Fix wrap-around %
 53                PPG_struct.PreviousTimestamp = PPG_struct.PreviousTimestamp - k_Max_Timestamp;
 54            end
 55            
 56            % The current datapoint is already one datapoint behind the peak
 57            % datapoint therefore the previoustimestamp is used
 58            PPG_struct.PulseRate = uint8(k_TimeStampsPerMin / (single(PPG_struct.PreviousTimestamp) - PPG_struct.PreviousQTime(1)));
 59            
 60            PPG_struct.PreviousQTime(1) = single(PPG_struct.PreviousTimestamp); %Interpolated_zerotime;
 61            PPG_struct.Quadrant = uint8(2);  % progress to next PPG_struct.Quadrant state in waveform
 62        end
 63    elseif (PPG_struct.FilteredDataInput < single(0.0)) 
 64        % signal must be low is it crossed zero again, reducing threshold...
 65        if (PPG_struct.Threshold > PPG_struct.Knob_Threshold_Goal) 
 66            PPG_struct.Threshold = PPG_struct.Threshold / single(1.2);
 67        elseif (PPG_struct.Threshold > 10.0) 
 68            PPG_struct.Threshold = PPG_struct.Threshold / single(1.02);
 69        end
 70    else 
 71        PPG_struct.PreviousInputVal = PPG_struct.FilteredDataInput;
 72    end
 73end
 74
 75if (PPG_struct.Quadrant == 2) 
 76    % peak systolic pressure has been detected, this section looks for the zero crossing
 77    % This PPG_struct.Quadrant cahnges from 2 to 3 when negative 
 78    % zero-crossing is detected. The calculated pulserate is reported as Q3
 79    % pulse rate
 80
 81    if (PPG_struct.FilteredDataInput < single(0.0)) 
 82        PPG_struct.Quadrant = uint8(3);                 % PEAK DETECTED move to next PPG_struct.Quadrant state
 83        if (PPG_struct.PreviousTimestamp > PPG_struct.Timestamp) 
 84            % Fix wrap-around %
 85            i0 = PPG_struct.PreviousTimestamp - k_Max_Timestamp;
 86
 87            PPG_struct.PreviousTimestamp = i0;
 88        end
 89
 90%       Interpolated_zerotime = single(PPG_struct.PreviousTimestamp) -  (PPG_struct.PreviousInputVal/(PPG_struct.FilteredDataInput - PPG_struct.PreviousInputVal)) *  single(PPG_struct.Timestamp - PPG_struct.PreviousTimestamp);
 91        Interpolated_zerotime = single(PPG_struct.PreviousTimestamp) +  (PPG_struct.PreviousInputVal/(PPG_struct.PreviousInputVal - PPG_struct.FilteredDataInput)) *  single(PPG_struct.Timestamp - PPG_struct.PreviousTimestamp);
 92
 93        if (PPG_struct.PreviousQTime(2) > Interpolated_zerotime)  % Fix wrap-around %
 94            PPG_struct.PreviousQTime(2) = PPG_struct.PreviousQTime(2) - single(k_Max_Timestamp);
 95        end
 96
 97        PPG_struct.PulseRate = uint8(k_TimeStampsPerMin/(Interpolated_zerotime - PPG_struct.PreviousQTime(2)));
 98
 99        PPG_struct.PreviousQTime(2) = Interpolated_zerotime;
100
101
102        %  Filter new peak value into Threshold. 
103        %     If large positive jump then attenuate the change.
104        if (PPG_struct.Knob_ThresholdTC > single(1.5) * PPG_struct.Threshold) 
105            f0 = PPG_struct.Knob_ThresholdTC/single(4.0);
106        else
107            f0 = PPG_struct.Knob_ThresholdTC;
108        end
109        
110        PPG_struct.Threshold = PPG_struct.Threshold * (single(1.0) - f0) + ((PPG_struct.LocalHigh - PPG_struct.LocalLow)/single(4.0)) * f0;   
111        PPG_struct.LocalLow = single(0.0);
112    
113    else 
114        PPG_struct.PreviousInputVal = PPG_struct.FilteredDataInput;
115    end
116end
117
118if (PPG_struct.Quadrant == 3) 
119    if (PPG_struct.FilteredDataInput < -PPG_struct.Threshold) 
120        if (PPG_struct.FilteredDataInput < PPG_struct.LocalLow) 
121            PPG_struct.LocalLow = PPG_struct.FilteredDataInput;
122        else 
123            % Valley detected
124            
125            PPG_struct.Quadrant = uint8(4);     % negative-slope zero cross followed by negative threshold crossing has been achieved - move to state 4
126                        
127            if (PPG_struct.PreviousTimestamp > PPG_struct.Timestamp) % Fix wrap-around %
128                PPG_struct.PreviousTimestamp = PPG_struct.PreviousTimestamp - k_Max_Timestamp;
129            end
130
131            PPG_struct.PulseRate = uint8(k_TimeStampsPerMin/(single(PPG_struct.PreviousTimestamp) - PPG_struct.PreviousQTime(3)));
132            PPG_struct.PreviousQTime(3) = single(PPG_struct.PreviousTimestamp); %Interpolated_zerotime;
133        end
134    elseif (PPG_struct.FilteredDataInput > single(0.0)) 
135            %signal must be low is it crossed zero again, reducing
136            %threshold...
137        if (PPG_struct.Threshold > PPG_struct.Knob_Threshold_Goal) 
138            PPG_struct.Threshold = PPG_struct.Threshold / single(1.2);
139        elseif (PPG_struct.Threshold > single(10.0)) 
140            PPG_struct.Threshold = PPG_struct.Threshold / single(1.02);
141        end
142    else 
143        PPG_struct.PreviousInputVal = PPG_struct.FilteredDataInput;
144    end
145end
146
147if (PPG_struct.Quadrant == 4) 
148    if (PPG_struct.FilteredDataInput > single(0.0)) 
149        PPG_struct.Quadrant = uint8(1);
150    
151        if (PPG_struct.PreviousTimestamp > PPG_struct.Timestamp) % Fix wrap-around %
152            PPG_struct.PreviousTimestamp = PPG_struct.PreviousTimestamp - k_Max_Timestamp;
153        end
154
155        Interpolated_zerotime = single(PPG_struct.PreviousTimestamp) + (PPG_struct.PreviousInputVal/(PPG_struct.PreviousInputVal - PPG_struct.FilteredDataInput) * single(PPG_struct.Timestamp - PPG_struct.PreviousTimestamp));
156
157        if (PPG_struct.PreviousQTime(4) > Interpolated_zerotime)    % Fix wrap-around %
158            PPG_struct.PreviousQTime(4) = PPG_struct.PreviousQTime(4) - single(k_Max_Timestamp);
159        end
160
161        PPG_struct.PulseRate = uint8(k_TimeStampsPerMin/(Interpolated_zerotime - PPG_struct.PreviousQTime(4)));
162        PPG_struct.PreviousQTime(4) = Interpolated_zerotime;
163
164
165        %  Filter new peak value into Threshold. 
166        %     If large positive jump then attenuate the change.
167        if (PPG_struct.Knob_ThresholdTC > single(1.5) * PPG_struct.Threshold) 
168            f0 = PPG_struct.Knob_ThresholdTC / single(4.0);
169        else
170            f0 = PPG_struct.Knob_ThresholdTC;
171        end 
172        
173        PPG_struct.Threshold = PPG_struct.Threshold * (single(1.0) - f0) + (PPG_struct.LocalHigh - PPG_struct.LocalLow)/single(4.0) * f0;     
174
175        PPG_struct.LocalHigh = single(0.0);
176
177    else 
178        PPG_struct.PreviousInputVal = PPG_struct.FilteredDataInput;
179    end
180end
181
182PPG_struct.PreviousTimestamp = PPG_struct.Timestamp;
183
184
185end % of function
